1107 private function isAllowed(string $action): bool
1108 {
1109 $group = GraphContext::$user->group;
1110
1111 // validate action
1112 // if action is one of the keys in the array self::SECURE_ACTIONS
1113 if (!array_key_exists($action, self::SECURE_ACTIONS)) {
1114 throw new RuntimeException("Action not defined in secure actions. Action: {$action}");
1115 }
1116
1117 // if is admin, allow all
1118 if ($group->id === 'admin') {
1119 return true;
1120 }
1121
1122 // if action is in the SAFE_ACTIONS, allow all
1123 if (self::SECURE_ACTIONS[$action]) {
1124 return true;
1125 }
1126
1127 // if action is restricted, only allow contributor
1128 if (self::SECURE_ACTIONS[$action] == false && $group->id == 'contributor')
1129 {
1130 return true;
1131 }
1132
1133 return false;
1134 }
